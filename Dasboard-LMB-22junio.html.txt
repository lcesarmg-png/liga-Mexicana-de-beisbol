function showTab(id){
document.querySelectorAll('.content').forEach(e=>e.classList.remove('active'));
document.querySelectorAll('.tab').forEach(e=>e.classList.remove('active'));
document.getElementById(id).classList.add('active');
event.target.classList.add('active');
if(id=='grafica') loadChart();
}
function filterZona(z){
document.querySelectorAll('#tablaNorte tr, #tablaSur tr').forEach((r,i)=>{
if(i==0) return;
if(z=='all') r.style.display='';
else if(z=='norte') r.style.display=r.classList.contains('norte')?'':'none';
else if(z=='sur') r.style.display=r.classList.contains('sur')?'':'none';
else if(z=='hoy') r.style.display=r.classList.contains('hoy')?'':'none';
});
}
